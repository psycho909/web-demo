clamp(MIN,VAL,MAX)
				如果VAL在MIN和MAX範圍之間，則使用VAL
				當VAL > MAX = MAX,當VAL < MIN = MIN
				width: clamp(400px, 50vw, 600px);

				max(VAL,MAX)
				當VAL > MAX = VAL
				width: max(50vw, 1000px);

				min(VAL,MIN)
				當VAL < MIN = VAL

MAX
